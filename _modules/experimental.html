[docs]def main():
    total_pairs = data.SplitPairs().test_pairs()
    pairs_batch = data.BatchData()
    batch = next(pairs_batch.batches(total_pairs, batch_size=6))

    indices = [[idx] for vals in batch.ids_map.values() for idx in range(vals, vals + settings.TOTAL_ROTATIONS)]

    print(batch.ids_map.keys())
    print(indices)

    indices_str = tf.placeholder(tf.float32, [None, 1])
    pairs_a = tf.placeholder(tf.int32, [None])
    pairs_b = tf.placeholder(tf.int32, [None])
    gathered_a = tf.gather(indices_str, pairs_a)
    gathered_b = tf.gather(indices_str, pairs_b)

    loss = tf.tanh(gathered_a - gathered_b)
    loss = tf.reduce_sum(loss)

    with tf.Session() as sess:
        sess.run(tf.global_variables_initializer())
        ret = sess.run([loss], feed_dict={
            indices_str: list(indices),
            pairs_a: batch.pairs_a,
            pairs_b: batch.pairs_b
        })
        print(ret)
